#include <iostream> 
using namespace std; 
 
int main() { 
    int choice; 
    int n = 0, students[100];  
    int rollNumber; 
    bool dataEntered = false; 
 
    do { 
        cout << "\n--- Training Program Attendance Menu ---\n"; 
        cout << "1. Enter roll numbers of students who attended\n"; 
        cout << "2. Check if a student attended\n"; 
        cout << "3. Exit\n"; 
        cout << "Enter your choice: "; 
        cin >> choice; 
 
        switch (choice) { 
        case 1: 
            cout << "Enter total number of students present: "; 
            cin >> n; 
            cout << "Enter roll numbers of students:\n"; 
            for (int i = 0; i < n; i++) { 
                cin >> students[i]; 
            } 
            dataEntered = true; 
            break; 
 
        case 2: 
            if (!dataEntered) { 
                cout << "Please enter attendance data first (Option 1).\n"; 
                break; 
            } 
            cout << "Enter roll number to check: "; 
            cin >> rollNumber; 
 
            { 
                bool found = false; 
                for (int i = 0; i < n; i++) { 
                    if (students[i] == rollNumber) { 
                        found = true; 
                        break; 
                    } 
                } 
                if (found) 
                    cout << "Student with Roll Number " << rollNumber << " was PRESENT.\n"; 
                else 
                    cout << "Student with Roll Number " << rollNumber << " was ABSENT.\n"; 
            } 
            break; 
 
        case 3: 
            cout << "Exiting program...\n"; 
            break; 
 
        default: 
            cout << "Invalid choice! Please try again.\n"; 
        } 
 
    } while (choice != 3); 
 
    return 0; 
}
