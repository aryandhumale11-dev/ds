#include <iostream> 
#include <string> 
using namespace std; 
 
const int TABLE_SIZE = 10; // Hash Table size 
 
struct Client { 
    string name; 
    string phone; 
    bool isOccupied; 
    Client() { 
        name = ""; 
        phone = ""; 
        isOccupied = false; 
    } 
}; 
 
// Hash function 
int hashFunction(string key) { 
    int sum = 0; 
    for (char c : key) sum += c; 
    return sum % TABLE_SIZE; 
} 
 
// Secondary hash for double hashing 
int secondaryHash(string key) { 
    int sum = 0; 
    for (char c : key) sum += c; 
    return 7 - (sum % 7); // must be smaller than table size 
} 
 
// Insert using Linear Probing 
void insertLinear(Client table[], string name, string phone) { 
    int index = hashFunction(name); 
    int originalIndex = index; 
    while (table[index].isOccupied) { 
        index = (index + 1) % TABLE_SIZE; 
        if (index == originalIndex) { 
            cout << "Hash table full!\n"; 
            return; 
        } 
    } 
    table[index].name = name; 
    table[index].phone = phone; 
    table[index].isOccupied = true; 
} 
 
// Insert using Quadratic Probing 
void insertQuadratic(Client table[], string name, string phone) { 
    int index = hashFunction(name); 
    int i = 0; 
    while (table[(index + i * i) % TABLE_SIZE].isOccupied) { 
        i++; 
        if (i == TABLE_SIZE) { 
            cout << "Hash table full!\n"; 
            return; 
        } 
    } 
    int finalIndex = (index + i * i) % TABLE_SIZE; 
    table[finalIndex].name = name; 
    table[finalIndex].phone = phone; 
    table[finalIndex].isOccupied = true; 
} 
 
// Insert using Double Hashing 
void insertDoubleHash(Client table[], string name, string phone) { 
    int index = hashFunction(name); 
    int step = secondaryHash(name); 
    while (table[index].isOccupied) { 
        index = (index + step) % TABLE_SIZE; 
    } 
    table[index].name = name; 
    table[index].phone = phone; 
    table[index].isOccupied = true; 
} 
 
// Search for a client 
void search(Client table[], string name, string method) { 
    int index = hashFunction(name); 
    int step = secondaryHash(name); 
    int i = 0; 
 
    while (i < TABLE_SIZE) { 
        int pos; 
        if (method == "linear") pos = (index + i) % TABLE_SIZE; 
        else if (method == "quadratic") pos = (index + i * i) % TABLE_SIZE; 
        else pos = (index + i * step) % TABLE_SIZE; 
 
        if (table[pos].isOccupied && table[pos].name == name) { 
            cout << "Found: " << table[pos].name << " -> " << table[pos].phone << "\n"; 
            return; 
        } 
        i++; 
    } 
    cout << "Client not found.\n"; 
} 
 
// Display table 
void display(Client table[]) { 
    for (int i = 0; i < TABLE_SIZE; i++) { 
        cout << i << ": "; 
        if (table[i].isOccupied) { 
            cout << table[i].name << " - " << table[i].phone; 
        } else { 
            cout << "Empty"; 
        } 
        cout << "\n"; 
    } 
} 
 
int main() { 
    Client tableLinear[TABLE_SIZE], tableQuadratic[TABLE_SIZE], tableDouble[TABLE_SIZE]; 
    int choice; 
    string name, phone; 
 
    do { 
        cout << "\n--- Telephone Book Menu ---\n"; 
        cout << "1. Insert (Linear Probing)\n"; 
        cout << "2. Insert (Quadratic Probing)\n"; 
        cout << "3. Insert (Double Hashing)\n"; 
        cout << "4. Search (Linear Probing)\n"; 
        cout << "5. Search (Quadratic Probing)\n"; 
        cout << "6. Search (Double Hashing)\n"; 
        cout << "7. Display (Linear Table)\n"; 
        cout << "8. Display (Quadratic Table)\n"; 
        cout << "9. Display (Double Hash Table)\n"; 
        cout << "0. Exit\n"; 
        cout << "Enter choice: "; 
        cin >> choice; 
 
        switch (choice) { 
        case 1: 
            cout << "Enter client name: "; 
            cin >> name; 
            cout << "Enter phone number: "; 
            cin >> phone; 
            insertLinear(tableLinear, name, phone); 
            break; 
        case 2: 
            cout << "Enter client name: "; 
            cin >> name; 
            cout << "Enter phone number: "; 
            cin >> phone; 
            insertQuadratic(tableQuadratic, name, phone); 
            break; 
        case 3: 
            cout << "Enter client name: "; 
            cin >> name; 
            cout << "Enter phone number: "; 
            cin >> phone; 
            insertDoubleHash(tableDouble, name, phone); 
            break; 
        case 4: 
            cout << "Enter client name to search: "; 
            cin >> name; 
            search(tableLinear, name, "linear"); 
            break; 
        case 5: 
            cout << "Enter client name to search: "; 
            cin >> name; 
            search(tableQuadratic, name, "quadratic"); 
            break; 
        case 6: 
            cout << "Enter client name to search: "; 
            cin >> name; 
            search(tableDouble, name, "double"); 
            break; 
        case 7: 
            display(tableLinear); 
            break; 
        case 8: 
            display(tableQuadratic); 
            break; 
        case 9: 
            display(tableDouble); 
            break; 
        case 0: 
            cout << "Exiting...\n"; 
            break; 
        default: 
            cout << "Invalid choice!\n"; 
        } 
    } while (choice != 0); 
 
    return 0; 
}
