#include <iostream> 
using namespace std; 
 
// Selection Sort 
void selectionSort(float arr[], int n) { 
    for (int i = 0; i < n - 1; i++) { 
        int minIndex = i; 
        for (int j = i + 1; j < n; j++) { 
            if (arr[j] < arr[minIndex]) 
                minIndex = j; 
        } 
        swap(arr[i], arr[minIndex]); 
    } 
} 
 
// Bubble Sort 
void bubbleSort(float arr[], int n) { 
    for (int i = 0; i < n - 1; i++) { 
        for (int j = 0; j < n - i - 1; j++) { 
            if (arr[j] > arr[j + 1]) 
                swap(arr[j], arr[j + 1]); 
        } 
    } 
} 
 
// Display Top 5 Scores 
void displayTopFive(float arr[], int n) { 
    cout << "\nTop Five Scores: "; 
    int count = 0; 
    for (int i = n - 1; i >= 0 && count < 5; i--) { 
        cout << arr[i] << " "; 
        count++; 
    } 
    cout << endl; 
} 
 
int main() { 
    int n; 
    cout << "Enter number of students: "; 
    cin >> n; 
 
    float percentages[100]; 
    cout << "Enter the percentages:\n"; 
    for (int i = 0; i < n; i++) { 
        cin >> percentages[i]; 
    } 
 
    // Selection Sort 
    float selectionArr[100]; 
    for (int i = 0; i < n; i++) selectionArr[i] = percentages[i]; 
    selectionSort(selectionArr, n); 
    cout << "\nPercentages after Selection Sort (Ascending): "; 
    for (int i = 0; i < n; i++) cout << selectionArr[i] << " "; 
    displayTopFive(selectionArr, n); 
 
    // Bubble Sort 
    float bubbleArr[100]; 
    for (int i = 0; i < n; i++) bubbleArr[i] = percentages[i]; 
    bubbleSort(bubbleArr, n); 
    cout << "\nPercentages after Bubble Sort (Ascending): "; 
    for (int i = 0; i < n; i++) cout << bubbleArr[i] << " "; 
    displayTopFive(bubbleArr, n); 
 
    return 0; 
}
