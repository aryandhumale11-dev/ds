#include <iostream> 
#include <vector> 
#include <queue> 
using namespace std; 
 
class Graph { 
    int V; // Number of cities 
    vector<vector<int>> adjList; // Adjacency list representation 
 
public: 
    Graph(int v) { 
        V = v; 
        adjList.resize(V); 
    } 
 
    // Add a flight path (edge) between two cities 
    void addEdge(int u, int v) { 
        adjList[u].push_back(v); 
        adjList[v].push_back(u); // Since flight paths are bidirectional 
    } 
 
    // Display adjacency list 
    void displayList() { 
        cout << "\nAdjacency List Representation:\n"; 
        for (int i = 0; i < V; i++) { 
            cout << "City " << i << " -> "; 
            for (auto x : adjList[i]) 
                cout << x << " "; 
            cout << endl; 
        } 
    } 
 
    // Convert to adjacency matrix and display 
    void displayMatrix() { 
        cout << "\nAdjacency Matrix Representation:\n"; 
        vector<vector<int>> matrix(V, vector<int>(V, 0)); 
 
        for (int i = 0; i < V; i++) { 
            for (auto x : adjList[i]) { 
                matrix[i][x] = 1; 
            } 
        } 
 
        for (int i = 0; i < V; i++) { 
            for (int j = 0; j < V; j++) 
                cout << matrix[i][j] << " "; 
            cout << endl; 
        } 
    } 
 
    // Check if the graph is connected using BFS 
    bool isConnected() { 
        vector<bool> visited(V, false); 
        queue<int> q; 
        int start = 0; 
        visited[start] = true; 
        q.push(start); 
 
        int count = 1; 
        while (!q.empty()) { 
            int node = q.front(); 
            q.pop(); 
 
            for (auto neigh : adjList[node]) { 
                if (!visited[neigh]) { 
                    visited[neigh] = true; 
                    q.push(neigh); 
                    count++; 
                } 
            } 
        } 
 
        return (count == V); 
    } 
}; 
 
int main() { 
    int V, E; 
    cout << "Enter number of cities: "; 
    cin >> V; 
 
    Graph g(V); 
 
    cout << "Enter number of flight paths: "; 
    cin >> E; 
 
    cout << "Enter the connected city pairs (A B):\n"; 
    for (int i = 0; i < E; i++) { 
        int a, b; 
        cin >> a >> b; 
        g.addEdge(a, b); 
    } 
 
    g.displayList(); 
    g.displayMatrix(); 
 
    if (g.isConnected()) 
        cout << "\nThe flight graph is CONNECTED.\n"; 
    else 
        cout << "\nThe flight graph is NOT CONNECTED.\n"; 
 
    return 0; 
}
