#include <iostream> 
#include <climits> 
using namespace std; 
void select_sort(int arr[], int size) { 
    int lowest_index; 
    for (int j = 0; j < size; j++) { 
        int lowest = INT_MAX; 
        for (int i = j; i < size; i++) { 
            if (lowest >= arr[i]) { 
                lowest = arr[i]; 
                lowest_index = i; 
            } 
        } 
        int temp = arr[j]; 
        arr[j] = arr[lowest_index]; 
        arr[lowest_index] = temp; 
    } 
} 
void binary_search(int arr[], int size, int search) { 
    int low = 0, high = size - 1; 
    while (low <= high) { 
        int mid = (low + high) / 2; 
        if (arr[mid] == search) { 
            cout << "Element found at position " << mid + 1 << endl; 
            return; 
        } else if (arr[mid] < search) { 
            low = mid + 1; 
        } else { 
            high = mid - 1; 
        } 
    } 
    cout << "Element not found" << endl; 
} 
int main() { 
    int size, search, task; 
    cout << "Input array size: "; 
    cin >> size; 
    int arr[100]; // Avoid VLA in C++, use fixed size array 
 
redo: 
    cout << "1] Input elements\n2] Sort\n3] Search\n:- "; 
    cin >> task; 
 
    switch (task) { 
    case 1: 
        for (int i = 0; i < size; i++) { 
            cout << "Input element " << i + 1 << ": "; 
            cin >> arr[i]; 
        } 
        // No break, so it continues to case 2 
    case 2: 
        select_sort(arr, size); 
        cout << "Sorted array: "; 
        for (int i = 0; i < size; i++) { 
            cout << arr[i] << " "; 
        } 
        cout << endl; 
        goto redo; 
    case 3: 
        cout << "Input number to search: "; 
        cin >> search; 
        binary_search(arr, size, search); 
        break; 
    default: 
        cout << "Invalid option" << endl; 
    } 
 
    return 0; 
}
