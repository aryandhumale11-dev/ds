#include <stdio.h> 
#include <stdlib.h> 
#include <string.h> 
 
#define MAX 10  // Maximum number of jobs in the queue 
 
typedef struct { 
    int id; 
    char name[30]; 
} Job; 
 
typedef struct { 
    Job jobs[MAX]; 
    int front, rear; 
} Queue; 
 
// Function prototypes 
void initialize(Queue *q); 
int isFull(Queue *q); 
int isEmpty(Queue *q); 
void addJob(Queue *q, int id, char name[]); 
void deleteJob(Queue *q); 
void displayJobs(Queue *q); 
 
// Initialize the queue 
void initialize(Queue *q) { 
    q->front = -1; 
    q->rear = -1; 
} 
 
// Check if queue is full 
int isFull(Queue *q) { 
    return q->rear == MAX - 1; 
} 
 
 
// Check if queue is empty 
int isEmpty(Queue *q) { 
    return q->front == -1 || q->front > q->rear; 
} 
 
// Add job to the queue 
void addJob(Queue *q, int id, char name[]) { 
    if (isFull(q)) { 
        printf("Queue is full! Cannot add more jobs.\n"); 
        return; 
    } 
 
    if (q->front == -1) q->front = 0; 
    q->rear++; 
    q->jobs[q->rear].id = id; 
    strcpy(q->jobs[q->rear].name, name); 
 
    printf("Job '%s' added successfully!\n", name); 
} 
 
// Delete job from the queue 
void deleteJob(Queue *q) { 
    if (isEmpty(q)) { 
        printf("No jobs to process! Queue is empty.\n"); 
        return; 
    } 
 
    printf("Processing and removing job: ID = %d, Name = %s\n", 
           q->jobs[q->front].id, q->jobs[q->front].name); 
    q->front++; 
 
    if (q->front > q->rear) { 
        q->front = q->rear = -1; // Reset queue when empty 
    } 
} 
 
// Display all jobs in the queue 
void displayJobs(Queue *q) { 
    if (isEmpty(q)) { 
        printf("No jobs in the queue.\n"); 
        return; 
    } 
 
    printf("\n--- Current Job Queue ---\n"); 
    for (int i = q->front; i <= q->rear; i++) { 
        printf("Job ID: %d, Name: %s\n", q->jobs[i].id, q->jobs[i].name); 
    } 
    printf("--------------------------\n"); 
} 
 
 
 
 
// Main function 
int main() { 
    Queue q; 
    initialize(&q); 
 
    int choice, id; 
    char name[30]; 
 
    do { 
        printf("\n--- Job Queue Menu ---\n"); 
        printf("1. Add Job\n"); 
        printf("2. Delete Job\n"); 
        printf("3. Display Jobs\n"); 
        printf("4. Exit\n"); 
        printf("Enter your choice: "); 
        scanf("%d", &choice); 
        getchar(); // to consume newline 
 
        switch (choice) { 
        case 1: 
            printf("Enter Job ID: "); 
            scanf("%d", &id); 
            getchar(); 
            printf("Enter Job Name: "); 
            fgets(name, sizeof(name), stdin); 
            name[strcspn(name, "\n")] = '\0'; // Remove newline 
            addJob(&q, id, name); 
            break; 
 
        case 2: 
            deleteJob(&q); 
            break; 
 
        case 3: 
            displayJobs(&q); 
            break; 
 
        case 4: 
            printf("Exiting... Thank you!\n"); 
            break; 
 
        default: 
            printf("Invalid choice! Try again.\n"); 
        } 
    } while (choice != 4); 
 
    return 0; 
}
